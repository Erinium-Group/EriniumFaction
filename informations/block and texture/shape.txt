public VoxelShape makeShape(){
	VoxelShape shape = Shapes.empty();
	shape = Shapes.join(shape, Shapes.box(0.0625, -0.625, 1.0625, 0.9375, 0, 1.9375), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0.0625, 0.25, 0.9375, 0.9375, 0.5625, 1.8125), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0.4375, 0.3125, 1.9375, 0.5625, 0.5625, 2), BooleanOp.OR);

	return shape;
}
